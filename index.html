jade.plugin

A jade plugin for flatiron

usage

var jade = require('jade.plugin');
app.use(jade.plugin, 
  {dir: __dirname + '/views', ext: '.jade' });

app.router.get('/', function(){
  app.render(this.res, 'index', {title: 'foo'});
});

install

npm install jade.plugin

test

npm test
